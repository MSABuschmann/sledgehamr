17 static void ChopGrids(amrex::BoxArray &ba, int target_size) {
18  const int max_grid_size = 8192;
19  const int blocking_factor = 16;
20 
21  amrex::IntVect chunk(max_grid_size, max_grid_size, max_grid_size);
22  chunk.min(ba.minimalBox().length());
23 
24  while (ba.size() < target_size) {
25  amrex::IntVect chunk_prev = chunk;
26 
27  std::array<std::pair<int, int>, AMREX_SPACEDIM> chunk_dir{AMREX_D_DECL(
28  std::make_pair(chunk[0], int(0)), std::make_pair(chunk[1], int(1)),
29  std::make_pair(chunk[2], int(2)))};
30  std::sort(chunk_dir.begin(), chunk_dir.end());
31 
32  for (int idx = AMREX_SPACEDIM - 1; idx >= 0; idx--) {
33  int idim = chunk_dir[idx].second;
34  int new_chunk_size = chunk[idim] / 2;
35  if (new_chunk_size != 0 && new_chunk_size % blocking_factor == 0) {
36  chunk[idim] = new_chunk_size;
37  ba.maxSize(chunk);
38  break;
39  }
40  }
41 
42  if (chunk == chunk_prev) {
43  break;
44  }
45  }
46 }
